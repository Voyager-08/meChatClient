cmake_minimum_required(VERSION 3.16)
project(meChatClient VERSION 0.1 LANGUAGES CXX)

# 设置默认构建类型为Release
if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE "Release" CACHE STRING "Choose the type of build, options are: Debug Release RelWithDebInfo MinSizeRel" FORCE)
endif()

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

# === Qt 路径（仅 Windows）===
if(WIN32)
    set(QT_ROOT "D:/QT/QT_x/6.5.3/mingw_64" CACHE PATH "Qt root")
    list(APPEND CMAKE_PREFIX_PATH "${QT_ROOT}")
endif()

find_package(Qt6 REQUIRED COMPONENTS Core Widgets Sql Network)

# === MySQL 路径（仅 Windows）===
if(WIN32)
    set(MYSQL_ROOT_DIR "D:/QT/QtFiles/MySQL" CACHE PATH "MySQL root")
    set(MYSQL_INCLUDE_DIR "${MYSQL_ROOT_DIR}/include")
    set(LIBMYSQL_DLL_PATH "${MYSQL_ROOT_DIR}/bin/libmysql.dll")
    set(LIBCRYPTO_DLL_PATH "${MYSQL_ROOT_DIR}/bin/libcrypto-3-x64.dll")
    set(LIBSSL_DLL_PATH "${MYSQL_ROOT_DIR}/bin/libssl-3-x64.dll")
    
    if(NOT EXISTS "${MYSQL_INCLUDE_DIR}/mysql.h")
        message(FATAL_ERROR "mysql.h not found!")
    endif()
    if(NOT EXISTS "${LIBMYSQL_DLL_PATH}")
        message(FATAL_ERROR "libmysql.dll not found!")
    endif()
endif()

# === 可执行文件 ===
add_executable(meChatClient
    src/main.cpp
    src/core/main/mainwindow.cpp
    src/core/main/mainwindow.h
    src/core/login/loginwindow.cpp
    src/core/login/loginwindow.h
    src/core/login/registerwidget.cpp
    src/core/login/registerwidget.h
    src/core/chat/chatwindow.cpp
    src/core/chat/chatwindow.h
    src/core/chat/chatwindow.ui
    src/core/extra/messagebubble.cpp
    src/core/extra/messagebubble.h
    src/sql/database.h
    src/sql/database.cpp
    src/thread/network/networkmanager.cpp
    src/thread/network/networkmanager.h
    src/thread/dataLoad/dataloader.cpp
    src/thread/dataLoad/dataloader.h
    src/custom/struct.h
    src/custom/meplaintextedit.cpp
    src/custom/meplaintextedit.h
    src/custom/clickablelabel.cpp
    src/custom/clickablelabel.h
    src/modelView/messagemodel.cpp
    src/modelView/messagemodel.h
    src/modelView/contactmodel.cpp
    src/modelView/contactmodel.h
    src/modelView/addcontactmodel.cpp
    src/modelView/addcontactmodel.h
    src/modelView/contactdelegate.cpp
    src/modelView/contactdelegate.h
    src/modelView/addcontactdelegate.cpp
    src/modelView/addcontactdelegate.h
    src/modelView/messagedelegate.cpp
    src/modelView/messagedelegate.h
    resources.qrc
)

# === 包含目录 ===
target_include_directories(meChatClient PRIVATE ${CMAKE_SOURCE_DIR})
if(WIN32)
    target_include_directories(meChatClient PRIVATE ${MYSQL_INCLUDE_DIR})
endif()

# === 链接库 ===
target_link_libraries(meChatClient PRIVATE Qt6::Core Qt6::Widgets Qt6::Sql Qt6::Network)

# 链接 MySQL 库
if(WIN32)
    target_link_libraries(meChatClient PRIVATE "${MYSQL_ROOT_DIR}/lib/libmysql.lib")
endif()

# === 输出目录：统一放到 <build-dir>/bin 下 ===
# 支持所有配置（Debug/Release/RelWithDebInfo...）
foreach(CONFIG_TYPE DEBUG RELEASE RELWITHDEBINFO MINSIZEREL)
    string(TOUPPER ${CONFIG_TYPE} UPPER_CONFIG)
    set_target_properties(meChatClient PROPERTIES
        RUNTIME_OUTPUT_DIRECTORY_${UPPER_CONFIG} ${CMAKE_BINARY_DIR}/bin
    )
endforeach()

# === 编译选项 ===
set_target_properties(meChatClient PROPERTIES
    COMPILE_DEFINITIONS_RELEASE "QT_NO_DEBUG_OUTPUT;QT_NO_INFO_OUTPUT;QT_NO_WARNING_OUTPUT;QT_NO_QDEBUG"
)
target_compile_definitions(meChatClient PRIVATE QT_DEPRECATED_WARNINGS)

# 添加MSVC编译选项以支持C++17和解决PDB文件访问错误
if(MSVC)
    target_compile_options(meChatClient PRIVATE /Zc:__cplusplus /permissive- /FS)
endif()

# 如果使用MinGW编译器且为Release模式
if(MINGW AND CMAKE_BUILD_TYPE STREQUAL "Release")
    # 设置Windows子系统，不显示控制台窗口
    set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -mwindows")
elseif(MSVC AND CMAKE_BUILD_TYPE STREQUAL "Release")
    # 如果使用MSVC编译器且为Release模式
    set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} /SUBSYSTEM:WINDOWS")
endif()


# === Windows: 自动复制运行时依赖 ===
if(WIN32)
    # 1. 复制 libmysql.dll 到 bin/
    add_custom_command(TARGET meChatClient POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "${LIBMYSQL_DLL_PATH}" 
            "$<TARGET_FILE_DIR:meChatClient>/libmysql.dll"
        COMMENT "Copy libmysql.dll to output dir"
    )

    # 2. 创建 sqldrivers 目录
    add_custom_command(TARGET meChatClient POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E make_directory
            "$<TARGET_FILE_DIR:meChatClient>/sqldrivers"
    )

    # 3. 根据实际构建配置（Debug/Release）复制对应的驱动
    set(QT_PLUGIN_DIR "${QT_ROOT}/plugins/sqldrivers")
    add_custom_command(TARGET meChatClient POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "${QT_PLUGIN_DIR}/qsqlmysql.dll"
            "$<TARGET_FILE_DIR:meChatClient>/sqldrivers/qsqlmysql.dll"
        COMMENT "Copy Qt MySQL driver"
    )
    add_custom_command(TARGET meChatClient POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "${QT_PLUGIN_DIR}/qsqlmysqld.dll"
            "$<TARGET_FILE_DIR:meChatClient>/sqldrivers/qsqlmysqld.dll"
        COMMENT "Copy Qt MySQL driver (Debug)"
    )
    add_custom_command(TARGET meChatClient POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "${QT_PLUGIN_DIR}/qsqlite.dll"
            "$<TARGET_FILE_DIR:meChatClient>/sqldrivers/qsqlite.dll"
        COMMENT "Copy Qt SQLite driver"
    )

    # 4. 复制平台插件
    add_custom_command(TARGET meChatClient POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E make_directory
            "$<TARGET_FILE_DIR:meChatClient>/platforms"
    )
    set(QWINDOWS_DLL "${QT_ROOT}/plugins/platforms/qwindows.dll")
    if(EXISTS "${QWINDOWS_DLL}")
        add_custom_command(TARGET meChatClient POST_BUILD
            COMMAND ${CMAKE_COMMAND} -E copy_if_different
                "${QWINDOWS_DLL}"
                "$<TARGET_FILE_DIR:meChatClient>/platforms/qwindows.dll"
            COMMENT "Copy qwindows.dll"
        )
    endif()
    
    # 5. 复制 libssl-3-x64.dll 和 libcrypto-3-x64.dll
    add_custom_command(TARGET meChatClient POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "${LIBSSL_DLL_PATH}"
            "$<TARGET_FILE_DIR:meChatClient>/libssl-3-x64.dll"
        COMMENT "Copy libssl-3-x64.dll to output dir"
    )

    add_custom_command(TARGET meChatClient POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "${LIBCRYPTO_DLL_PATH}"
            "$<TARGET_FILE_DIR:meChatClient>/libcrypto-3-x64.dll"
        COMMENT "Copy libcrypto-3-x64.dll to output dir"
    )

    # 6. 复制 MinGW 运行时（可选）
    if(CMAKE_COMPILER_IS_GNUCXX OR MINGW)
        foreach(rt_dll libgcc_s_seh-1.dll libstdc++-6.dll libwinpthread-1.dll)
            set(src "${QT_ROOT}/bin/${rt_dll}")
            if(EXISTS "${src}")
                add_custom_command(TARGET meChatClient POST_BUILD
                    COMMAND ${CMAKE_COMMAND} -E copy_if_different
                        "${src}"
                        "$<TARGET_FILE_DIR:meChatClient>/${rt_dll}"
                    COMMENT "Copy ${rt_dll}"
                )
            endif()
        endforeach()
    endif()

    # 7. 复制 Qt 核心库（开发环境需要）
    set(QT_BIN_DIR "${QT_ROOT}/bin")
    foreach(qt_dll Qt6Core.dll Qt6Gui.dll Qt6Widgets.dll Qt6Sql.dll Qt6Network.dll)
        if(EXISTS "${QT_BIN_DIR}/${qt_dll}")
            add_custom_command(TARGET meChatClient POST_BUILD
                COMMAND ${CMAKE_COMMAND} -E copy_if_different
                    "${QT_BIN_DIR}/${qt_dll}"
                    "$<TARGET_FILE_DIR:meChatClient>/${qt_dll}"
                COMMENT "Copy ${qt_dll}"
            )
        endif()
    endforeach()
endif()

# 在构建时创建 images/avatar 目录
if(WIN32)
    add_custom_command(TARGET meChatClient POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E make_directory
            "$<TARGET_FILE_DIR:meChatClient>/images/avatar"
        COMMENT "Creating images/avatar directory for user avatars"
    )
endif()
