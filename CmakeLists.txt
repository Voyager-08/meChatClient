cmake_minimum_required(VERSION 3.16)
project(weChat VERSION 0.1 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

# === Qt 路径（仅 Windows）===
if(WIN32)
    set(QT_ROOT "D:/QT/QT_x/6.5.3/mingw_64" CACHE PATH "Qt root")
    list(APPEND CMAKE_PREFIX_PATH "${QT_ROOT}")
endif()

find_package(Qt6 REQUIRED COMPONENTS Core Widgets Sql Network)

# === MySQL 路径（仅 Windows）===
if(WIN32)
    set(MYSQL_ROOT_DIR "D:/QT/QtFiles/MySQL" CACHE PATH "MySQL root")
    set(MYSQL_INCLUDE_DIR "${MYSQL_ROOT_DIR}/include")
    set(LIBMYSQL_DLL_PATH "${MYSQL_ROOT_DIR}/bin/libmysql.dll")

    if(NOT EXISTS "${MYSQL_INCLUDE_DIR}/mysql.h")
        message(FATAL_ERROR "mysql.h not found!")
    endif()
    if(NOT EXISTS "${LIBMYSQL_DLL_PATH}")
        message(FATAL_ERROR "libmysql.dll not found!")
    endif()
endif()

# === 可执行文件 ===
add_executable(weChat
    src/main.cpp
    src/login/loginwindow.cpp
    src/login/loginwindow.h
    src/login/registerwidget.cpp
    src/login/registerwidget.h
    src/sql/database.h
    src/sql/database.cpp
    src/chat/chatwindow.cpp
    src/chat/chatwindow.h
    src/chat/chatwindow.ui
    src/chat/contactmessagebubble.cpp
    src/chat/contactmessagebubble.h
    src/chat/messagebubble.cpp
    src/chat/messagebubble.h
    src/chat/networkmanager.cpp
    src/chat/networkmanager.h
    src/custom/meplaintextedit.cpp
    src/custom/meplaintextedit.h
    src/custom/clickablelabel.cpp
    src/custom/clickablelabel.h
    src/contact/contactmodel.cpp
    src/contact/contactmodel.h
    src/contact/contactdelegate.cpp
    src/contact/contactdelegate.h
    resources.qrc
)

# === 包含目录 ===
target_include_directories(weChat PRIVATE ${CMAKE_SOURCE_DIR})
if(WIN32)
    target_include_directories(weChat PRIVATE ${MYSQL_INCLUDE_DIR})
endif()

# === 链接库 ===
target_link_libraries(weChat PRIVATE Qt6::Core Qt6::Widgets Qt6::Sql Qt6::Network)

# === 输出目录：统一放到 <build-dir>/bin 下 ===
# 支持所有配置（Debug/Release/RelWithDebInfo...）
foreach(CONFIG_TYPE DEBUG RELEASE RELWITHDEBINFO MINSIZEREL)
    string(TOUPPER ${CONFIG_TYPE} UPPER_CONFIG)
    set_target_properties(weChat PROPERTIES
        RUNTIME_OUTPUT_DIRECTORY_${UPPER_CONFIG} ${CMAKE_BINARY_DIR}/bin
    )
endforeach()

# === 编译选项 ===
set_target_properties(weChat PROPERTIES
    COMPILE_DEFINITIONS_RELEASE "QT_NO_DEBUG_OUTPUT;QT_NO_INFO_OUTPUT;QT_NO_WARNING_OUTPUT"
)
target_compile_definitions(weChat PRIVATE QT_DEPRECATED_WARNINGS)

# === Windows: 自动复制运行时依赖 ===
if(WIN32)
    # 1. 复制 libmysql.dll 到 bin/
    add_custom_command(TARGET weChat POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "${LIBMYSQL_DLL_PATH}"
            "$<TARGET_FILE_DIR:weChat>/libmysql.dll"
        COMMENT "Copy libmysql.dll to output dir"
    )

    # 2. 创建 sqldrivers 目录
    add_custom_command(TARGET weChat POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E make_directory
            "$<TARGET_FILE_DIR:weChat>/sqldrivers"
    )

    # 3. 根据实际构建配置（Debug/Release）复制对应的驱动
    set(QT_PLUGIN_DIR "${QT_ROOT}/plugins/sqldrivers")
    add_custom_command(TARGET weChat POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "$<$<CONFIG:Debug>:${QT_PLUGIN_DIR}/qsqlmysqld.dll>$<$<NOT:$<CONFIG:Debug>>:${QT_PLUGIN_DIR}/qsqlmysql.dll>"
            "$<TARGET_FILE_DIR:weChat>/sqldrivers/$<$<CONFIG:Debug>:qsqlmysqld.dll>$<$<NOT:$<CONFIG:Debug>>:qsqlmysql.dll>"
        COMMENT "Copy Qt MySQL driver for $<CONFIG> config"
    )

    # 4. 复制平台插件
    add_custom_command(TARGET weChat POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E make_directory
            "$<TARGET_FILE_DIR:weChat>/platforms"
    )
    set(QWINDOWS_DLL "${QT_ROOT}/plugins/platforms/qwindows.dll")
    if(EXISTS "${QWINDOWS_DLL}")
        add_custom_command(TARGET weChat POST_BUILD
            COMMAND ${CMAKE_COMMAND} -E copy_if_different
                "${QWINDOWS_DLL}"
                "$<TARGET_FILE_DIR:weChat>/platforms/qwindows.dll"
            COMMENT "Copy qwindows.dll"
        )
    endif()

    # 5. 复制 MinGW 运行时（可选）
    if(CMAKE_COMPILER_IS_GNUCXX OR MINGW)
        foreach(rt_dll libgcc_s_seh-1.dll libstdc++-6.dll libwinpthread-1.dll)
            set(src "${QT_ROOT}/bin/${rt_dll}")
            if(EXISTS "${src}")
                add_custom_command(TARGET weChat POST_BUILD
                    COMMAND ${CMAKE_COMMAND} -E copy_if_different
                        "${src}"
                        "$<TARGET_FILE_DIR:weChat>/${rt_dll}"
                    COMMENT "Copy ${rt_dll}"
                )
            endif()
        endforeach()
    endif()
endif()
